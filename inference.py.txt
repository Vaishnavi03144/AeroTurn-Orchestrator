import asyncio
import os
import textwrap
import json
import re
from typing import List, Optional
from openai import OpenAI

# Import your local logic
from env import AeroTurnEnv
from models import Action

# Mandatory Variables from Environment
API_KEY = os.getenv("HF_TOKEN")
API_BASE_URL = os.getenv("API_BASE_URL") or "https://router.huggingface.co/v1"
MODEL_NAME = os.getenv("MODEL_NAME") or "meta-llama/Llama-3-70b-instruct"

# Metadata for Logging
TASK_NAME = "aeroturn-optimization"
BENCHMARK = "aviation-ground-ops-v1"
MAX_STEPS = 5

def log_start(task: str, env_name: str, model: str) -> None:
    print(f"[START] task={task} env={env_name} model={model}", flush=True)

def log_step(step: int, action: str, reward: float, done: bool, error: Optional[str]) -> None:
    error_val = error if error else "null"
    done_val = str(done).lower()
    print(f"[STEP] step={step} action={action} reward={reward:.2f} done={done_val} error={error_val}", flush=True)

def log_end(success: bool, steps: int, score: float, rewards: List[float]) -> None:
    rewards_str = ",".join(f"{r:.2f}" for r in rewards)
    print(f"[END] success={str(success).lower()} steps={steps} score={score:.2f} rewards={rewards_str}", flush=True)

def extract_json(text):
    try:
        match = re.search(r'\{.*\}', text, re.DOTALL)
        return json.loads(match.group(0)) if match else None
    except: return None

async def main() -> None:
    client = OpenAI(base_url=API_BASE_URL, api_key=API_KEY)
    env = AeroTurnEnv() # Local Environment
    
    # We will run Task 3 (Hard) for the final inference evaluation
    obs = env.reset(task_id=3)
    
    history_rewards = []
    steps_taken = 0
    total_reward = 0.0
    
    log_start(task=TASK_NAME, env=BENCHMARK, model=MODEL_NAME)

    try:
        for step in range(1, MAX_STEPS + 1):
            # 1. Build Prompt
            prompt = f"Time: {obs.time_step}. Resources: {obs.resources.json()}. Fleet: {obs.fleet}. Output JSON with 'callsign', 'command', 'gate_id'."
            
            # 2. Get Model Decision
            response = client.chat.completions.create(
                model=MODEL_NAME,
                messages=[{"role": "system", "content": "You are an ATC Lead."}, {"role": "user", "content": prompt}],
                temperature=0.2
            )
            
            raw_content = response.choices[0].message.content
            action_dict = extract_json(raw_content)
            
            if action_dict:
                # 3. Apply Step
                action_obj = Action(**action_dict)
                obs, reward, done, info = env.step(action_obj)
                
                error = None
                action_str = f"{action_obj.command}_{action_obj.callsign}"
            else:
                reward = 0.0
                done = False
                error = "json_parse_error"
                action_str = "no_action"

            # 4. Mandatory Step Logging
            history_rewards.append(reward)
            total_reward += reward
            steps_taken = step
            log_step(step=step, action=action_str, reward=reward, done=done, error=error)

            if done:
                break

        # Final Score Calculation (Normalized 0 to 1)
        final_score = min(max(total_reward / 2.0, 0.0), 1.0)
        success = final_score > 0.5

    except Exception as e:
        print(f"[DEBUG] Error: {e}")
        success = False
        final_score = 0.0
    finally:
        log_end(success=success, steps=steps_taken, score=final_score, rewards=history_rewards)

if __name__ == "__main__":
    asyncio.run(main())
