from fastapi import FastAPI, Request
from env import AeroTurnEnv
from models import Action
import uvicorn

app = FastAPI()
env = AeroTurnEnv()

@app.get("/")
async def read_root():
    return {"status": "AeroTurn Orchestrator is Live"}

@app.post("/reset")
async def reset(request: Request):
    try:
        body = await request.json()
    except:
        body = {}
    
    # We must AWAIT the async reset call now
    return await env.reset(**body)

@app.post("/step")
async def step(action: Action):
    # We must AWAIT the async step call now
    obs, reward, done, info = await env.step(action)
    return {
        "observation": obs,
        "reward": reward,
        "done": done,
        "info": info
    }

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=7860)
