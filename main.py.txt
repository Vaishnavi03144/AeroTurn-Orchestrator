from fastapi import FastAPI
from env import AeroTurnEnv  # Fixed name
from models import Action

app = FastAPI()
env = AeroTurnEnv()  # Fixed name

@app.get("/")
def read_root():
    return {"status": "AeroTurn Orchestrator is Live"}

@app.post("/reset")
def reset():
    return env.reset()

@app.post("/step")
def step(action: Action):
    obs, reward, done, info = env.step(action)
    return {
        "observation": obs,
        "reward": reward,
        "done": done,
        "info": info
    }
