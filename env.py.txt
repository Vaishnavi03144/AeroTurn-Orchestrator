import json
# Remove the gymnasium import entirely
from models import Observation, Aircraft, Action

class AeroTurnEnv: # Remove (gym.Env) - standard Python class is fine here
    def __init__(self):
        # super().__init__() # Remove this line too
        self.planes = []

    def reset(self, seed=None, options=None):
        # Initial scenario with conflict
        # Note: Added seed/options to match standard gym signature
        self.planes = [
            Aircraft(callsign="AI101", altitude=300, heading=90),
            Aircraft(callsign="AI202", altitude=300, heading=270)
        ]
        observation = Observation(radar=self.planes, task_description="Separate aircraft altitudes.")
        return observation, {} # Return obs and empty info dict

    def step(self, action: Action):
        # Update altitude for the specific aircraft
        # Ensure 'action' is handled as an object or dict as per your 'models.py'
        for p in self.planes:
            if p.callsign == action.callsign:
                p.altitude = action.target_altitude
        
        # Scoring logic: Reward if altitudes are different
        alts = [p.altitude for p in self.planes]
        
        # We need a clear separation (e.g., at least 100ft) for "Heavy" anime-level safety
        reward = 1.0 if abs(alts[0] - alts[1]) >= 100 else 0.0
        
        # Done if we have successfully separated them
        terminated = True if reward == 1.0 else False
        truncated = False
        
        observation = Observation(radar=self.planes, task_description="")
        
        return observation, reward, terminated, truncated, {}
