import json
from models import Observation, Aircraft, Action

class ATCEnv:
    def __init__(self):
        self.reset()

    def reset(self):
        # Scenario: Two planes at same altitude (Conflict!)
        self.planes = [
            Aircraft(callsign="AI101", altitude=300, heading=90),
            Aircraft(callsign="AI202", altitude=300, heading=270)
        ]
        return Observation(radar=self.planes, task_description="Separate aircraft altitudes.")

    def step(self, action: Action):
        # Update altitude
        for p in self.planes:
            if p.callsign == action.callsign:
                p.altitude = action.target_altitude
        
        # Check if they are separated (Score logic)
        alts = [p.altitude for p in self.planes]
        reward = 1.0 if len(set(alts)) > 1 else 0.0
        done = True if reward == 1.0 else False
        
        return Observation(radar=self.planes, task_description=""), reward, done, {}
